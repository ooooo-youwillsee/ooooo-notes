# create new dir
cd /sys/fs/cgroup
mkdir test

# creat loop.sh for testing cpu quota
vim loop.sh
while :
do
        :
done

# lunch loop.sh, generate pid -> 2584068
nohup sh loop.sh &

# echo pid to cgroup.procs
echo 2584068 > test/cgroup.procs

# set cpu, at lease 0.1
echo 1000 10000 > test/cpu.max

# check 
top 

# recovery all
kill 2584068
rmdir test
